    <script>
        let workbook = null;
        let worksheet = null;

        function handleFileUpload(event) {
            const file = event.target.files[0];
            const reader = new FileReader();

            reader.onload = function(e) {
                const data = new Uint8Array(e.target.result);
                workbook = XLSX.read(data, {type: 'array'});
                
                worksheet = workbook.Sheets[workbook.SheetNames[0]];
                
                const jsonData = XLSX.utils.sheet_to_json(worksheet, {header: 1});
                
                const tableBody = document.getElementById('tableBody');
                tableBody.innerHTML = '';
                
                for (let i = 1; i < jsonData.length; i++) {
                    const row = jsonData[i];
                    const tr = document.createElement('tr');
                    
                    const avatarCell = document.createElement('td');
                    avatarCell.textContent = row[0] || '';
                    
                    const nameCell = document.createElement('td');
                    nameCell.textContent = row[1] || '';
                    
                    const attendanceCell = document.createElement('td');
                    attendanceCell.textContent = row[2] || 0;
                    attendanceCell.id = `attendance-${i}`;
                    
                    const actionCell = document.createElement('td');
                    const incrementBtn = document.createElement('button');
                    incrementBtn.textContent = 'Increment';
                    incrementBtn.onclick = () => incrementAttendance(i, attendanceCell);
                    actionCell.appendChild(incrementBtn);
                    
                    tr.appendChild(avatarCell);
                    tr.appendChild(nameCell);
                    tr.appendChild(attendanceCell);
                    tr.appendChild(actionCell);
                    
                    tableBody.appendChild(tr);
                }
            };

            reader.readAsArrayBuffer(file);
        }

        function incrementAttendance(rowIndex, attendanceCell) {
            let currentAttendance = parseInt(attendanceCell.textContent) || 0;
            
            currentAttendance++;
            
            attendanceCell.textContent = currentAttendance;
            
            if (worksheet) {
                const cellAddress = XLSX.utils.encode_cell({r: rowIndex, c: 2});
                worksheet[cellAddress] = { v: currentAttendance };
            }
        }

        function exportToExcel() {
            if (!workbook) {
                alert('Please upload a file first');
                return;
            }

            XLSX.writeFile(workbook, 'updated_check_in.xlsx');
        }
